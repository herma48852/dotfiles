* INITIALIZATION & NATIVE COMP
This section dynamically detects the Homebrew path and the latest GCC version to fix the "libgccjit" and "gcc driver" errors[cite: 1].
It works on both Intel (x64) and Apple Silicon (M-series)[cite: 2, 4].

#+BEGIN_SRC emacs-lisp
  ;; Silence the "zsh" migration warning globally for all Emacs shells 
  (setenv "BASH_SILENCE_DEPRECATION_WARNING" "1")
  
  ;; Standard "Panic" key for canceling commands
  (global-set-key (kbd "C-g") 'keyboard-quit)
#+END_SRC

#+BEGIN_SRC emacs-lisp
(when (eq system-type 'darwin)
  ;; 1. Ensure Emacs has the correct PATH and environment from the shell 
  (use-package exec-path-from-shell
    :ensure t
    :config
    (when (memq window-system '(mac ns x))
      (exec-path-from-shell-initialize)
      (exec-path-from-shell-copy-env "LIBRARY_PATH")
      ;; Also copy the Gemini API key from your .bash_profile
      (exec-path-from-shell-copy-env "GEMINI_API_KEY")))

  ;; 2. Robust GCC/libgccjit path discovery for Native Compilation [cite: 4, 5]
  (let* ((brew-bin (if (file-exists-p "/opt/homebrew/bin/brew") 
                       "/opt/homebrew/bin/brew"   ; Apple Silicon Path 
                     "/usr/local/bin/brew"))      ; Intel Path 
         (gcc-prefix (when (file-exists-p brew-bin)
                       (string-trim (shell-command-to-string (concat brew-bin " --prefix gcc")))))
         (gcc-bin-dir (when gcc-prefix (concat gcc-prefix "/bin")))
         ;; Verify directory exists before searching for GCC executables 
         (gcc-bin (when (and gcc-bin-dir (file-directory-p gcc-bin-dir))
                    (directory-files gcc-bin-dir t "^gcc-[0-9]+$")))
         (gcc-executable (when gcc-bin (car (last gcc-bin))))
         (gcc-version (when gcc-executable 
                        (car (last (split-string gcc-executable "-"))))))
    
    (when (and gcc-executable gcc-version)
      ;; Set the driver path so Emacs can compile Lisp to machine code [cite: 6]
      (setq native-comp-driver-path gcc-executable)
      
      ;; Construct the LIBRARY_PATH specifically for libgccjit [cite: 7]
      (let ((lib-path (concat gcc-prefix "/lib/gcc/" gcc-version))
            (current-lib-path (or (getenv "LIBRARY_PATH") "")))
        (setenv "LIBRARY_PATH" 
                (if (string-match-p (regexp-quote lib-path) current-lib-path)
                    current-lib-path
                  (concat lib-path ":" current-lib-path)))))))
#+END_SRC

* Misc Key-Bindings
#+BEGIN_SRC emacs-lisp
  (windmove-default-keybindings)
  (setq windmove-wrap-around t)

  (use-package which-key
    :ensure t
    :config
       (which-key-mode))

  (defun buffer-or-filename-to-kill-ring ()
    "Store buffer file name or buffer name if not visting a file to kill-ring"
    (interactive)
    (cond ((buffer-file-name)
           (setq tmp-name (format "%s" (buffer-file-name))))
          (t (setq tmp-name (format "%s" (buffer-name)))))
    (split-window-below nil)
    (windmove-down nil)
    (switch-to-buffer "*scratch*" nil 'force-same-window)
    (end-of-buffer nil)
    (newline nil 1)
    (insert tmp-name)
    (set-mark-command nil)
    (move-beginning-of-line 1)
    (kill-ring-save 1 2 1)
    (delete-window))

  (defun create-named-shell ()
    (interactive)
    (shell)
    (rename-buffer
     (read-from-minibuffer "Rename-buffer (to new name): ")))

  (defun reload-init-el ()
    (interactive)
    (load-file "~/.emacs.d/init.el"))

  (global-set-key (kbd "C-c r r") `reload-init-el)
  (global-set-key (kbd "C-c f f") `buffer-or-filename-to-kill-ring)
  (global-set-key (kbd "C-x x s") `create-named-shell)
  (global-set-key [f9] `compile)
  (global-set-key (kbd "C-x x r") 'rename-buffer)
  (global-superword-mode t)
  (global-set-key (kbd "C-x p") 'find-file-at-point)

* Orderless
#+BEGIN_SRC emacs-lisp
;; The `orderless' package lets the minibuffer use an out-of-order
;; pattern matching algorithm.  It matches space-separated words or
;; regular expressions in any order.  In its simplest form, something
;; like "ins pac" matches `package-menu-mark-install' as well as
;; `package-install'.  This is a powerful tool because we no longer
;; need to remember exactly how something is named.
;;
;; Note that Emacs has lots of "completion styles" (pattern matching
;; algorithms), but let us keep things simple.
;;
;; Further reading: https://protesilaos.com/emacs/dotemacs#h:7cc77fd0-8f98-4fc0-80be-48a758fcb6e2
(use-package orderless
  :ensure t
  :config
  (setq completion-styles '(orderless basic)))

#+END_SRC
* Consult
#+BEGIN_SRC emacs-lisp
;; The `consult' package provides lots of commands that are enhanced
;; variants of basic, built-in functionality.  One of the headline
;; features of `consult' is its preview facility, where it shows in
;; another Emacs window the context of what is currently matched in
;; the minibuffer.  Here I define key bindings for some commands you
;; may find useful.  The mnemonic for their prefix is "alternative
;; search" (as opposed to the basic C-s or C-r keys).
;;
;; Further reading: https://protesilaos.com/emacs/dotemacs#h:22e97b4c-d88d-4deb-9ab3-f80631f9ff1d
(use-package consult
  :ensure t
  :bind (;; A recursive grep
         ("M-s M-g" . consult-grep)
         ;; Search for files names recursively
         ("M-s M-f" . consult-find)
         ;; Search through the outline (headings) of the file
         ("M-s M-o" . consult-outline)
         ;; Search the current buffer
         ("M-s M-l" . consult-line)
         ;; Switch to another buffer, or bookmarked file, or recently
         ;; opened file.
         ("M-s M-b" . consult-buffer)))
#+END_SRC
